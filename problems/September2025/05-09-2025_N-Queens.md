# N-Queens (LeetCode)
Link: https://leetcode.com/problems/n-queens/description/

Solution - 
## Backtracking
- **Time Complexity**: O(N!)
- **Space Complexity**: O(N)
```C++
class Solution {
private:
    vector<vector<string>> res;
    vector<string> board;
    vector<bool> visited;
     bool canPlace(int row, int col) {
        // 1. Check for a queen on the same row to the left.
        for (int c = 0; c < col; ++c) {
            if (board[row][c] == 'Q') {
                return false;
            }
        }

        // 2. Check for a queen on the upper-left diagonal.
        for (int r = row, c = col; r >= 0 && c >= 0; --r, --c) {
            if (board[r][c] == 'Q') {
                return false;
            }
        }
        
        // 3. Check for a queen on the lower-left diagonal.
        for (int r = row, c = col; r < board.size() && c >= 0; ++r, --c) {
            if (board[r][c] == 'Q') {
                return false;
            }
        }
        
        return true;
    }
    void nQueen(int col) {
        if(col >= board.size()) {
            res.push_back(board);
            return;
        }
        for(int row = 0; row < board.size(); row++) {
             if(canPlace(row, col)) {
                board[row][col] = 'Q';
                nQueen(col+1);
                board[row][col] = '.';
            }
        }
    }
public:
    vector<vector<string>> solveNQueens(int n) {
        board.resize(n, string(n,'.'));
        nQueen(0);
        return res;
    }
};
```

## Optimised Backtracking
### check for safe place in O(1)
- **Time Complexity**: O(N!)
- **Space Complexity**: O(N)
```C++
class Solution {
private:
    vector<vector<string>> res;
    vector<string> board;
    vector<bool> visited;
    vector<bool> rows;
    vector<bool> diag1;
    vector<bool> diag2;
    void nQueen(int col) {
        int n = board.size();
        if(col >= n) {
            res.push_back(board);
            return;
        }
        for(int row = 0; row < n; row++) {
             if(!rows[row] && !diag1[row+col] && !diag2[row-col+n]) {
                rows[row] = diag1[row+col] = diag2[row-col+n] = true;
                board[row][col] = 'Q';
                nQueen(col+1);
                board[row][col] = '.';
                rows[row] = diag1[row+col] = diag2[row-col+n] = false;
            
            }
        }
    }
public:
    vector<vector<string>> solveNQueens(int n) {
        board.resize(n, string(n,'.'));
        rows.resize(n+1, false);
        diag1.resize(2*n+1, false);
        diag2.resize(2*n+1, false);
        nQueen(0);
        return res;
    }
};
```
